日期字符串要to_date后再枚举转数组

	('2012-1-1'...'2012-2-2').to_a
	=> ["2012-1-1",
	 "2012-1-2",
	 "2012-1-3",
	 "2012-1-4",
	 "2012-1-5",
	 "2012-1-6",
	 "2012-1-7",
	 "2012-1-8",
	 "2012-1-9",
	 "2012-2-0",
	 "2012-2-1"]

	 ('2012-1-1'.to_date...'2012-2-2'.to_date).to_a
